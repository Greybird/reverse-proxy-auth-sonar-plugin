21      /**
22       * Tests the typical authentication process.
23       */
24      @Test
25      public void testRealm() {
26          final Settings settings = new Settings();
27          settings.appendProperty("reverseproxyauth.header.name",
28                  "X-Forwarded-User");
29          settings.appendProperty("reverseproxyauth.localhost", "localhost");
30  
31          final SecurityRealm realm = new ReverseProxyAuthRealm(settings);
32          realm.getName();
33          realm.init();
34  
35          final HttpServletRequest httpServletRequest = mock(HttpServletRequest.class);
36          when(httpServletRequest.getHeader("X-Forwarded-User")).thenReturn(
37                  "foo@bar.com");
38          when(httpServletRequest.getServerName())
39                  .thenReturn("not.localhost.com");
40  
41          final ExternalUsersProvider.Context usersProviderContext = new ExternalUsersProvider.Context(
42                  null, httpServletRequest);
43          realm.getUsersProvider().doGetUserDetails(usersProviderContext);
44  
45          final Authenticator.Context authenticatorContext = new Authenticator.Context(
46                  null, null, httpServletRequest);
47          realm.doGetAuthenticator().doAuthenticate(authenticatorContext);
48      }
