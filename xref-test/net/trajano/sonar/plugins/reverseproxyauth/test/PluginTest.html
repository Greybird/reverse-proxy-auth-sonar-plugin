17      public void testExtensionProviderWhenRealmIsNotSet() {
18          final Settings settings = new Settings();
19  
20          final List<?> list = new Extensions(settings).provide();
21          Assert.assertTrue(list.isEmpty());
22      }
23  
24      @Test
25      public void testExtensionProviderWhenRealmIsSet() {
26          final Settings settings = new Settings();
27          settings.appendProperty("sonar.security.realm",
28                  ReverseProxyAuthPlugin.KEY);
29  
30          final List<Class<? extends Extension>> list = new Extensions(settings)
31                  .provide();
32          Assert.assertTrue(list.contains(ReverseProxyAuthRealm.class));
33          Assert.assertTrue(list.contains(ValidateRedirectionFilter.class));
34      }
